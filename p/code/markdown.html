Octant Finder
def get_octant(x,y):
    try:
        if abs(x)/abs(y) == 1:
            return "undefined"
    except:
        return "undefined"
    dx, dy = x,y
    octant = 0
    if dy < 0:
        dx, dy = -dx, -dy  # rotate by 180 degrees
        octant += 4
    if dx < 0:
        dx, dy = dy, -dx  # rotate clockwise by 90 degrees
        octant += 2
    if dx < dy:
        # no need to rotate now
        octant += 1
    return octant
